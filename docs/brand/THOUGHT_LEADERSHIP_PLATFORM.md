#### Event ROI Measurement

```javascript
// Event ROI Calculator
class EventROICalculator {
    calculateEventROI(event) {
        const costs = {
            registration: event.registrationFee,
            travel: event.travelCosts,
            accommodation: event.accommodationCosts,
            materials: event.marketingMaterials,
            time: event.timeInvestment
        };

        const benefits = {
            leads: event.leadsGenerated * 250, // R250 per qualified lead
            partnerships: event.partnerships * 5000, // R5000 per partnership
            media_coverage: event.mediaMentions * 1000, // R1000 per mention
            brand_awareness: event.audienceReach * 0.5, // R0.5 per person reached
            speaking_fees: event.speakingFees || 0
        };

        const totalCosts = Object.values(costs).reduce((a, b) => a + b, 0);
        const totalBenefits = Object.values(benefits).reduce((a, b) => a + b, 0);

        return {
            totalCosts,
            totalBenefits,
            roi: (totalBenefits - totalCosts) / totalCosts,
            breakEven: totalCosts / 250 // Number of leads needed to break even
        };
    }
}
```

// Thought Leadership ROI Calculator
class ThoughtLeadershipROI {
    calculateROI(thoughtLeadershipActivities) {
        const costs = {
            content_creation: thoughtLeadershipActivities.contentCosts,
            event_participation: thoughtLeadershipActivities.eventCosts,
            pr_and_marketing: thoughtLeadershipActivities.prCosts,
            team_time: thoughtLeadershipActivities.teamCosts,
            tools_and_platforms: thoughtLeadershipActivities.platformCosts
        };

        const benefits = {
            lead_generation: thoughtLeadershipActivities.leads * 500, // R500 per lead
            brand_awareness: thoughtLeadershipActivities.awarenessValue,
            customer_acquisition: thoughtLeadershipActivities.customers * 2000, // R2000 per customer
            partnership_opportunities: thoughtLeadershipActivities.partnerships * 10000, // R10000 per partnership
            media_coverage_value: thoughtLeadershipActivities.mediaValue
        };

        const totalCosts = Object.values(costs).reduce((a, b) => a + b, 0);
        const totalBenefits = Object.values(benefits).reduce((a, b) => a + b, 0);

        return {
            totalCosts,
            totalBenefits,
            netBenefit: totalBenefits - totalCosts,
            roi: (totalBenefits - totalCosts) / totalCosts,
            paybackPeriod: totalCosts / (totalBenefits / 12) // months
        };
    }
}
